```py


class Solution:
    def isHappy(self, n: int) -> bool:
        def next_num(n):
            total_sum=0
            while (n>0):
                digit=n%10
                total_sum += digit**2
                n=n//10
            return total_sum         //计算每个位的平方和
            
        seen=set()
        while n!=1 and n not in seen:       //判断n是否为1或重复出现
            seen.add(n)
            n=next_num(n)
        return n==1

```
